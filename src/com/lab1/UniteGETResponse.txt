      test_resp();
    }, 100));

  test_resp();

  function test_resp() {
    $('html .main_slider .slide').each(function() {
      if($('.main_slider .lazy').length > 0) {
        if(Math.abs($(this).position().left) <= parseInt($(window).width())) {
          if ($('body').width() < 769) {
            $(this).find('.lazy').attr('src',$(this).find('.lazy').attr('narrow')).removeClass('lazy').addClass('resp_imp');
          } else {
            $(this).find('.lazy').attr('src',$(this).find('.lazy').attr('wide')).removeClass('lazy').addClass('resp_imp');;
          }
        }
      }
    })
  }

  function switch_images_slide() {
    if ($('body').width() < 769) {
      $('.slide_resp').each(function() {
        $(this).attr('src', $(this).attr('narrow'));
      })
    } else {
      $('.slide_resp').each(function() {
        $(this).attr('src', $(this).attr('wide'));
      })
    }
  }

  $(window).resize(function() {
    clearTimeout($.data(this, 'scrollTimer'));
    $.data(this, 'scrollTimer', setTimeout(function() {
      switch_images_slide();
    }, 200));

  })

})
